def LIS_naive(A):
    n = len(A)
    B = [0 for i in range(n)]
    C = [0 for i in range(n)]

    for i in range(n):
        C[i] = 1
        B[i] = i
        for j in range(i-1):
            if A[j] < A[i] and C[j] >= C[i]:
                C[i] = C[j] + 1
                B[i] = j
    # Trace back to find the LIS
    ans = [] # Empty array for answer
    # First find the max value and index in C
    best_j = C.index(max(C))
    while best_j != B[best_j]: # If found index itself => reach the begin
        ans.append(A[best_j])
        best_j = B[best_j]
    # Now add the first element.
    ans.append(A[best_j])
    return ans[::-1] # return reverse array for output

A = [3, 1, 4, 1, 5, 9, 2, 6, 5, 3, 5]
print(LIS_naive(A))

import bisect
def LIS_dp(A):
    inf = int(1e9)
    n = len(A)
    B = [inf for i in range(n+1)]
    for i in range(n):
        j = bisect.bisect_left(B, A[i], 0, n) # aka find upper bound.
        if B[j] != A[j]:
            B[j] = A[i]

    ans = []
    for i in B:
        if i == inf:
            break    
        ans.append(i)
    return ans
print(LIS_dp(A))

2. Bài toán xếp balo 0-1 (0-1 knapsack)¶

t vn : Bi ton ny l g?

Bi ton a ra yu cu chn mt tp hp cc vt phm c gi tr v khi lng khc nhau t vo mt balo c khi lng gii hn. Mi vt phm c th c chn hoc khng c chn, v mi vt phm ch c chn mt ln.

V d: Cho mt balo c khi lng gii hn l 10 v danh sch cc vt phm nh sau:

  • Vt phm 1: c gi tr l 6 v khi lng l 2
  • Vt phm 2: c gi tr l 10 v khi lng l 5
  • Vt phm 3: c gi tr l 12 v khi lng l 6
  • Vt phm 4: c gi tr l 8 v khi lng l 3

Bi ton yu cu tm cch chn mt tp hp cc vt phm t vo balo, sao cho tng gi tr ca cc vt phm ny l ln nht v tng khi lng ca chng khng vt qu khi lng gii hn ca balo (tc l 10).

Kt qu ca bi ton ny l tp hp cc vt phm c chn, trong trng hp ny l vt phm 1, 2 v 4, vi tng gi tr = 24 v tng khi lng = 10.

t bi ton: Cho 2 mng $V$ v $W$ c di $n$, c nh s t $0$ n $n-1$. Mng $V$ lu gi tr ca cc vt phm, mng $W$ lu khi lng ca cc vt phm. Cho mt s $m$ l khi lng gii hn ca balo.

Cch gii:

  • Phng php ngy th (brute force):
    • To ra tt c cc tp con c th c ca tp hp cc vt phm. Vi mi tp con, tnh tng gi tr v tng khi lng ca cc vt phm trong tp con . Nu tng khi lng nh hn hoc bng $m$ v tng gi tr ln hn gi tr hin ti, th cp nht gi tr hin ti bng tng gi tr ca tp con .

def knapsack_naive(V, W, m):
    n = len(V)
    ans_subset = []
    ans_value = 0
    for i in range(2**n):
        subset = []
        value = 0
        weight = 0
        for j in range(n):
            if i & (1 << j):
                subset.append(j)
                value += V[j]
                weight += W[j]
        if weight <= m and value > ans_value:
            ans_value = value
            ans_subset = subset
    return ans_value, ans_subset

V = [6, 10, 12, 8]
W = [2, 5, 6, 3]
m = 10
print(knapsack_naive(V, W, m))

  • Phng php quy hoch ng:
    • To ra mt mng 2 chiu $A$ c kch thc $n \times m$. Ti v tr $A[i][j]$, ta lu gi tr ln nht c th t c vi $i$ vt phm u tin v khi lng gii hn l $j$.
    • Khi to gi tr ban u ca mng $A$ l $0$.
    • Duyt qua ln lt cc vt phm t $0$ n $n$. vt phm th $i$, ta duyt qua ln lt cc khi lng t $1$ n $m$ (bng bin $j$). Ti mi v tr $A[i][j]$, ta thc hin tnh ton nh sau:
      • Nu $W[i] > j$ (khi lng ca vt phm ln hn khi lng gii hn), th $A[i][j] = A[i-1][j]$ (gi tr ln nht c th t c vi $i-1$ vt phm v khi lng gii hn l $j$).
      • Nu $W[i] \leq j$ (khi lng ca vt phm nh hn hoc bng khi lng gii hn), th $A[i][j] = max(A[i-1][j], A[i-1][j-W[i]] + V[i])$ (gi tr ln nht c th t c vi $i-1$ vt phm v khi lng gii hn l $j$, hoc gi tr ln nht c th t c vi $i-1$ vt phm v khi lng gii hn l $j-W[i]$ cng vi gi tr ca vt phm th $i$).
    • Kt qu ca bi ton l $A[n][m]$ (gi tr ln nht c th t c vi $n$ vt phm v khi lng gii hn l $m$).

def knapsack_dp(V, W, m):
    n = len(V)
    dp = [[0 for j in range(m+1)] for i in range(n+1)]

    # Build table for dp from bottom up
    for i in range(n+1):
        for j in range(m+1):
            if i == 0 or j == 0:
                dp[i][j] = 0
            elif W[i-1] <= j:
                dp[i][j] = max(V[i-1] + dp[i-1][j-W[i-1]], dp[i-1][j])
            else:
                dp[i][j] = dp[i-1][j]

    # Answer of knapsack
    res = dp[n][m] 
    # Trace back to find the subset
    subset = []
    j = m
    for i in range(n, 0, -1):
        if res <= 0:
            break
        if res == dp[i-1][j]:
            continue
        else:
            subset.append(i-1)
            res = res - V[i-1]
            j = j - W[i-1]
    return dp[n][m], subset[::-1]

print(knapsack_dp(V, W, m))

nh ngha: Quy hoch ng ch s l phng php hiu qu gii cc bi ton m s tho mn tnh cht v cc ch s ca n.

t vn : C bao nhiu s $x$ trong khong $[a, b]$ tho mn 1 iu kin no .

y chnh l c s ca quy hoch ng. Mi bi ton m nh trn u c th c gii bng quy ch s (dp digit, hoc dp ch s.)

tng chnh:

  • Cho s $x$ c $n$ ch s. tng chnh ca dp ch s l biu din cc ch s ca $x$ thnh mt mng $t[]$. Gi s ta c $t_nt_{n-1}t_{n-2}...t_2t_1$ l biu din thp phn ca $x$ vi $t_i$ (0 < i <= n) l ch s th $i$ t phi sang. Ch s bn tri nht $t_n$ l ch s c trng s ln nht.

  • Gi, sau khi biu din s cho theo cch ny, ta sinh ra cc s nh hn s cho v ng thi tnh ton bng DP, nu s tho mn tnh cht cho. Ta bt u sinh ra cc s c s ch s = 1 v sau ln lt n khi s ch s = n. Cc s c s ch s t hn n c th c phn tch bng cch t cc ch s bn tri nht l 0.

V d: Cho 2 s nguyn $a$ v $b$. In ra tng cc ch s xut hin trong cc s nguyn trong khong $[a, b] V d: $a = 5$ v $b = 11$, kt qu l $38$ $(5 + 6 + 7 + 8 + 9 + 1 + 0 + 1 + 1)$ Rng buc: $1 \leq a \leq b \leq 10^{18}$

Now we see that if we have calculated the answer for state having n-1 digits, i.e., tn-1 tn-2 t2 t1 and we need to calculate answer for state having n digits tn tn-1 tn-2 t2 t1. So, clearly, we can use the result of the previous state instead of re-calculating it. Hence, it follows the overlapping property. Lets think for a state for this DP Our DP state will be dp(idx, tight, sum)

gii quyt bi ton, ta tnh xem trong trng hp c $n-1$ ch s ($t_{n-1}t_{n-2}...t_2t_1$) th $n$ ch s ($t_nt_{n-1}t_{n-2}...t_2t_1$) gii quyt nh th no?

Bi ton s c gii bng cch s dng bi ton con (trng hp trc ) thay v tnh li. V vy, n s c tnh cht trng lp.

Tnh gi tr newTight nh th no?

The new tight value from a state depends on its previous state. If tight value form the previous state is 1 and the digit at idx chosen is digitidx , then only our new tight will be 1 as it only then tells that the number formed till now is prefix of the limiting integer.

Gi tr newTight ca 1 trng thi ph thuc vo trng thi trc . Nu tight trc = 1 v ch s ti ch s idx c chn l digit[idx] (tc l ch s ti ch s idx trong s nguyn gii hn), th gi tr newTight l 1, v ch khi n cho bit rng s c to ra cho n nay l tin t ca s nguyn gii hn.

# Memoization for the state results
dp = [[[-1 for i in range(2)] for j in range(180)]for k in range(20)]

# Stores the digits in x in a list digit
def getDigits(x, digit):
	while x:
		digit.append(x % 10)
		x //= 10
		
# Return sum of digits from 1 to integer in digit list
def digitSum(index, sumof, tight, digit):

	# Base case
	if index == -1:
		return sumof
	
		# Checking if already calculated this state
	if dp[index][sumof][tight] != -1 and tight != 1:
		return dp[index][sumof][tight]
	ret = 0
	
	# Calculating range value
	k = digit[index] if tight else 9
	for i in range(0, k+1):
	
		# Calculating newTight value for nextstate
		newTight = tight if digit[index] == i else 0
		
		# Fetching answer from next state
		ret += digitSum(index-1, sumof+i, newTight, digit)
	if not tight:
		dp[index][sumof][tight] = ret
	return ret

# Returns sum of digits in numbers from a to b
def rangeDigitSum(a, b):
	digitA = []
	
	# Storing digits of a-1 in digitA
	getDigits(a-1, digitA)
	
	# Finding sum of digits from 1 to "a-1" which is passed as digitA
	ans1 = digitSum(len(digitA)-1, 0, 1, digitA)
	digitB = []
	
	# Storing digits of b in digitB
	getDigits(b, digitB)
	
	# Finding sum of digits from 1 to "b" which is passed as digitB
	ans2 = digitSum(len(digitB)-1, 0, 1, digitB)
	return ans2-ans1


a, b = 123, 1024
print("digit sum for given range: ", rangeDigitSum(a, b))
